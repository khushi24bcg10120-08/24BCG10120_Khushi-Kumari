1. [JOINS] List the name of each employee along with their job title and department name.


SELECT e.name, j.job_title, d.dept_name
FROM Employees e
JOIN Jobs j ON e.job_id = j.job_id
JOIN Departments d ON e.dept_id = d.dept_id;


2.[ORDER BY] Find the top 3 highest-paid employees and list them in descending order.


SELECT *
FROM Employees
ORDER BY salary DESC
LIMIT 3;


3. [GROUP BY] Find the number of employees hired each year


SELECT YEAR(hire_date), COUNT(*)
FROM Employees
GROUP BY YEAR(hire_date);


4. [JOINS] Find the job title and salary of the employee with ID 504.


SELECT j.job_title, e.salary
FROM Employees e
JOIN Jobs j
ON e.job_id = j.job_id
WHERE e.emp_id = 504;


5.[COMPLEX FILTER] Find employees who were hired between 2018 and 2020 and earn more than 60,000


SELECT *
FROM Employees
WHERE hire_date BETWEEN '2018-01-01' AND '2020-12-31'
AND salary > 60000;


6. [SELF JOIN] Retrieve a list of employee names and their respective manager's names.


SELECT e.name AS Employee,
m.name AS Manager
FROM Employees e
LEFT JOIN Employees m
ON e.manager_id = m.emp_id;


7. [GROUP BY] Calculate the total salary expenditure for each department name.


SELECT d.dept_name, SUM(e.salary)
FROM Employees e
JOIN Departments d
ON e.dept_id = d.dept_id
GROUP BY d.dept_name;


8. [MULTI-TABLE JOIN] Show all department names even if they have no employees assigned to them.


SELECT d.dept_name, e.name
FROM Departments d
LEFT JOIN Employees e
ON d.dept_id = e.dept_id;


9. [JOINS + WHERE] Find all Developers working in the 'San Francisco' location.


SELECT e.name
FROM Employees e
JOIN Jobs j ON e.job_id = j.job_id
JOIN Departments d ON e.dept_id = d.dept_id
WHERE j.job_title='Developer'
AND d.location='San Francisco';


10. [AGGREGATION] Find the difference between the highest and lowest salary in the company.


SELECT MAX(salary)-MIN(salary)
FROM Employees;


11. [SUBQUERY] Find the names of employees who work in the same department as 'David'.


SELECT name
FROM Employees
WHERE dept_id =
(
SELECT dept_id
FROM Employees
WHERE name='David'
);


12. [GROUP BY] For each manager (by ID), find the minimum salary of the people reporting to them.


SELECT manager_id,
MIN(salary)
FROM Employees
WHERE manager_id IS NOT NULL
GROUP BY manager_id;


13. [CORRELATED SUBQUERY] Find employees who earn more than the average salary of their own department.


SELECT *
FROM Employees e
WHERE salary >
(
SELECT AVG(salary)
FROM Employees
WHERE dept_id = e.dept_id
);


14. [HAVING] Identify managers (by ID) who manage more than 2 people.


SELECT manager_id
FROM Employees
WHERE manager_id IS NOT NULL
GROUP BY manager_id
HAVING COUNT(*) > 2;




15. [JOIN + ORDER BY] Display employee names and hire dates, sorted by department name (A-Z) and then by salary (High to Low).


SELECT e.name, e.hire_date
FROM Employees e
JOIN Departments d
ON e.dept_id = d.dept_id
ORDER BY d.dept_name ASC, e.salary DESC;


16. [SUBQUERY] Find the employee(s) with the third-highest salary.


SELECT *
FROM Employees
WHERE salary =
(
SELECT DISTINCT salary
FROM Employees
ORDER BY salary DESC
LIMIT 1 OFFSET 2
);



17. [HAVING] Display departments that have more than 2 employees.


SELECT d.dept_name,
COUNT(*) AS Total_Employees
FROM Employees e
JOIN Departments d
ON e.dept_id = d.dept_id
GROUP BY d.dept_name
HAVING COUNT(*) > 2;


18. [LEFT JOIN] List all job titles and the number of employees currently holding that job.


SELECT j.job_title, COUNT(e.emp_id)
FROM Jobs j
LEFT JOIN Employees e
ON j.job_id = e.job_id
GROUP BY j.job_title;


19. [JOIN + LIKE] Find employees in the 'Finance' department whose names contain the letter 'i'.


SELECT e.name
FROM Employees e
JOIN Departments d
ON e.dept_id = d.dept_id
WHERE d.dept_name='Finance'
AND e.name LIKE '%i%';
